Example 2: Batch Prediction
  import requests
  
  messages = [
    {"text": "Free money!"},
    {"text": "Hi there"},
    {"text": "URGENT: Claim prize!"}
  ]
  
  response = requests.post(
    "http://127.0.0.1:8000/batch-predict",
    json=messages
  )
  data = response.json()
